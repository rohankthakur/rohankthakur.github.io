from bs4 import BeautifulSoup
import requests
import pandas as pd
import numpy as np
import re

# set the url to scrape from ...
url = "https://en.wikipedia.org/wiki/List_of_states_and_union_territories_of_India_by_population"

# get contents from the url ... 
results = requests.get(url)
content = results.content
soup = BeautifulSoup(content, 'lxml')
#print(soup.prettify)
test = soup.find("table", {"class": "wikitable sortable"})

# dynamically detect row & column counts, and column labels
col_lbls = []
row_cnt = 0
col_cnt = 0
for row in test.find_all("tr"):
    td_tags = row.find_all("td")
    if len(td_tags) > 0:
        row_cnt += 1
        if col_cnt == 0:
            col_cnt = len(td_tags)
    # determine column labels if available
    th_tags = row.find_all("th")
    if len(th_tags) > 0 and len(col_lbls) == 0:
        for th in th_tags:
            #col_lbls.append(th.get_text())
            col_lbls.append(re.sub('[^a-zA-Z]+', '', th.get_text().title()))

# columns: check for label vs count mismatch
if len(col_lbls) != col_cnt:
    print('mismatch between column labels and count')
    col_used = range(0,col_cnt)
else:
    col_used = col_lbls

        col_count = 0
        for cell in cells:
            raw_table.iat[row_count, col_count] = cell.text.strip()

        row_count += 1
